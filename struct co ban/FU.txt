

USE HR_DB;
 


CREATE TABLE dbo.Regions(
  region_id     INT
    CONSTRAINT reg_region_id  PRIMARY KEY,
  region_name   VARCHAR(50)
)



CREATE TABLE dbo.Countries(
  country_id    CHAR(2)
    CONSTRAINT ctr_country_id PRIMARY KEY,
  country_name  VARCHAR(40)
    CONSTRAINT ctr_country_name_nn NOT NULL,
  region_id    INT
    CONSTRAINT ctr_reg_fk REFERENCES dbo.Regions(region_id)
)



CREATE TABLE dbo.Locations(
  location_id   INT
    CONSTRAINT loc_location_id  PRIMARY KEY,
  street_address  VARCHAR(50),
  postal_code     CHAR(12),
  city            VARCHAR(50),
  state_province  VARCHAR(50),
  country_id      CHAR(2)
    CONSTRAINT loc_ctr_fk REFERENCES dbo.Countries(country_id)
)



CREATE TABLE dbo.Jobs(
  job_id  VARCHAR(10)
    CONSTRAINT job_joib_id  PRIMARY KEY,
  job_title VARCHAR(100)
    CONSTRAINT job_job_title_nn NOT NULL,
  min_salary  NUMERIC(10, 2), 
  max_salary  NUMERIC(10, 2)
)


CREATE TABLE dbo.Departments(
  department_id   INT 
    CONSTRAINT dept_department_id PRIMARY KEY,
  department_name VARCHAR(100),
  manager_id      INT,
  location_id     INT
    CONSTRAINT dept_loc_fk REFERENCES dbo.Locations(location_id)
)


CREATE TABLE dbo.Employees(
  employee_id   INT
                CONSTRAINT emp_employee_id PRIMARY KEY,
  first_name    VARCHAR(20),
  last_name     VARCHAR(30) CONSTRAINT emp_last_name NOT NULL,
  email         VARCHAR(50)
                CONSTRAINT emp_email_nn NOT NULL
                CONSTRAINT emp_email_uk UNIQUE,
  phone_number  VARCHAR(20),
  hire_date     DATE
                CONSTRAINT emp_hire_date_nn NOT NULL,
  job_id        VARCHAR(10)
                CONSTRAINT emp_job_id_nn NOT NULL
                CONSTRAINT emp_job_fk REFERENCES dbo.Jobs(job_id),
  salary        NUMERIC(10, 2)
                CONSTRAINT emp_salary_ck CHECK (salary > 0),
  commission_pct  NUMERIC(6,2), -- commission percentage
  manager_id    INT,
  department_id INT,
    CONSTRAINT  emp_dept_fk FOREIGN KEY(department_id)  REFERENCES dbo.Departments(department_id) 
)

ALTER TABLE dbo.Employees
ALTER COLUMN  first_name   NVARCHAR(20)

/*
ALTER Employee table
*/
-- Add constraint
ALTER TABLE Employees
ADD CONSTRAINT emp_manager_id_fk FOREIGN KEY (manager_id) REFERENCES dbo.Employees(employee_id);

-- Add constraint to Department table
ALTER TABLE Departments
ADD CONSTRAINT dept_manager_id_fk FOREIGN KEY(manager_id) REFERENCES Employees(employee_id);


CREATE TABLE dbo.Job_History(
  employee_id     INT,
  [start_date]      DATE,
  end_date        DATE NOT NULL,
  job_id          VARCHAR(10),
  department_id   INT,
  CONSTRAINT job_his_ck CHECK(start_date < end_date),
  CONSTRAINT job_his_pk PRIMARY KEY (employee_id, start_date),
  CONSTRAINT job_his_job_fk FOREIGN KEY(job_id) REFERENCES dbo.Jobs(job_id),
  CONSTRAINT job_his_emp_fk FOREIGN KEY(employee_id) REFERENCES dbo.Employees(employee_id),
  CONSTRAINT job_his_dept_fk FOREIGN KEY (department_id) REFERENCES dbo.Departments(department_id)
)

SELECT TB1.job_id, job_title, TB1.salary FROM dbo.Jobs
join (
	SELECT job_id, salary FROM dbo.Employees
	where employee_id = manager_id
	ORDER BY salary DESC) as TB1
on TB1.job_id = Jobs.job_id;

-- 5
SELECT *  FROM dbo.Employees
	where max_salary - min_salary = 8000
	ORDER BY job_title ASC;
    
-- 6
SELECT department_name, employee_count FROM dbo.Departments
join (
	SELECT department_id, COUNT(employee_id) AS employee_count
	FROM dbo.Employees
	GROUP BY department_id) as TB1
on TB1.department_id = Departments.department_id;
